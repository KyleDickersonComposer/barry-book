% book-examples.tex — numbered musical examples and list of examples (LoE).
%
%   List of examples (LoE) ........... custom list like ToC but file extension .loe
%   Example heading + counter ........ \beginBookExample / \finishBookExample
%   Draft chapter banner ............. \incompleteContentNotice
% =============================================================================

\ExplSyntaxOn
\bool_new:N \g__bookexamples_example_active_bool

\msg_new:nnnn { bookexamples } { begin-example-while-active }
  { bookexamples:~beginBookExample~called~while~an~example~is~already~active. }
  {
    Call~\token_to_str:N \finishBookExample~before~starting~another~example.^^J
    Nested~example~wrappers~are~not~supported.
  }
\msg_new:nnnn { bookexamples } { finish-example-without-begin }
  { bookexamples:~finishBookExample~called~without~a~matching~beginBookExample. }
  {
    Start~an~example~with~\token_to_str:N \beginBookExample~before~calling~
    \token_to_str:N \finishBookExample.
  }

% -----------------------------------------------------------------------------
% List of Musical Examples (LoE): parallel to \tableofcontents but uses .loe file.
% \@starttoc{loe} reads/writes bin/main.loe (or alongside jobname). Entries use
% \addcontentsline{loe}{bookexample}{...} from \finishBookExample.
% \l@bookexample defines how each LoE line is formatted (dotted leader, indent).
% -----------------------------------------------------------------------------
\cs_new_protected:cpn { l@bookexample } #1#2
  { \use:c { @dottedtocline } { 1 } { 0em } { 2.3em } {#1} {#2} }

\newcommand{\listofmusicalexamples}{%
  \frontMatterChapter{List~of~Musical~Examples}%
  \addcontentsline{toc}{chapter}{List~of~Musical~Examples}%
  \use:c { @starttoc } { loe }%
}


% -----------------------------------------------------------------------------
% Numbered examples and LoE lines.
%
% \beginBookExample{Caption text}
%   ... music or prose ...
% \finishBookExample
%
% Hyperref / PDF semantics: \addcontentsline is deferred to \finishBookExample so the
% LoE page number points where the example *ends* (often better for long excerpts).
% The PDF link target must stay on the *heading*: we \xdef\bookExampleLoEHref at the
% start (copy of \@currentHref right after \phantomsection), then before \addcontentsline
% we temporarily restore \@currentHref inside a group so hyperref writes the LoE
% hyperlink to the heading anchor, not the end of the body.
% -----------------------------------------------------------------------------
\newlength{\bookExampleOuterSep}
\setlength{\bookExampleOuterSep}{1em}

\newlength{\bookExampleInnerSep}
\setlength{\bookExampleInnerSep}{1.35em}

\newlength{\bookExampleBottomSep}
\setlength{\bookExampleBottomSep}{2\bookExampleOuterSep}

\newcounter{bookExample}
\newcommand{\bookExampleLoEentry}{}
\newcommand{\bookExampleLoEHref}{}

\newcommand{\beginBookExample}[1]{%
  \bool_if:NT \g__bookexamples_example_active_bool
    { \msg_error:nn { bookexamples } { begin-example-while-active } }%
  \bool_gset_true:N \g__bookexamples_example_active_bool
  \refstepcounter{bookExample}%
  \phantomsection
  \xdef\bookExampleLoEHref{\cs:w @currentHref \cs_end:}%
  \gdef\bookExampleLoEentry{Example\space\arabic{bookExample}.\space#1}%
  \par\addvspace{\bookExampleOuterSep}%
  \noindent{\bfseries Example\space\arabic{bookExample}:\space}#1\par
  \nobreak\addvspace{\bookExampleInnerSep}%
}

\newcommand{\finishBookExample}{%
  \bool_if:NTF \g__bookexamples_example_active_bool
    {
      \par\addvspace{\bookExampleBottomSep}%
      \begingroup
        \expandafter\edef\csname @currentHref\endcsname{\bookExampleLoEHref}%
        \addcontentsline{loe}{bookexample}{\bookExampleLoEentry}%
      \endgroup
      \bool_gset_false:N \g__bookexamples_example_active_bool
    }
    { \msg_error:nn { bookexamples } { finish-example-without-begin } }
}
\ExplSyntaxOff

% -----------------------------------------------------------------------------

  \color{barryBlack}%
  \begin{minipage}{0.88\textwidth}%
    \rule{\linewidth}{0.4pt}\par
    \vspace{0.65em}%
    {\bfseries Draft notice.} This chapter is incomplete and part of an open source textbook in progress. Contributions, corrections, examples, and source-level improvements are welcome.
    \vspace{0.65em}\par
    \rule{\linewidth}{0.4pt}%
  \end{minipage}%
  \endgroup
  \par\vspace{0.5em}%
}
